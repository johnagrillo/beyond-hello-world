#include "simple.h"

#include <malloc.h>

#include <string.h>

struct Simple{

char * stringParam;

char * stringDefault;

char * stringParamRw;

int intParam;

int intDefault;

float floatParam;

float floatDefault;

};

struct Simple*

simple_new(char * stringParam,

char * stringParamRw,

int intParam,

float floatParam)

{

struct Simple * this = (struct Simple*)malloc(sizeof(struct Simple));

this->stringParam = stringParam;

this->stringDefault = "string default";

this->stringParamRw = stringParamRw;

this->intParam = intParam;

this->intDefault = 0;

this->floatParam = floatParam;

this->floatDefault = 0.0;

return this;

}

// free as a void; used in collections

void simple_free_v(void * v)

{

simple_free((struct Simple *) v);

}

void simple_free(struct Simple * this)

{

free(this);

}

char*

simple_getStringParam(struct Simple * this)

{

return this->stringParam;

}

char*

simple_getStringParamRw(struct Simple * this)

{

return this->stringParamRw;

}

struct Simple *

simple_setStringParamRw(struct Simple * this,

char * stringParamRw)

{

if (this->stringParamRw != NULL) free(this->stringParamRw);

this->stringParamRw=stringParamRw;

return this;

}

int

simple_getIntParam(struct Simple * this)

{

return this->intParam;

}

float

simple_getFloatParam(struct Simple * this)

{

return this->floatParam;

}

char *

simple_getKey(char * stringParam,

char * stringDefault,

char * stringParamRw,

int intParam,

int intDefault,

float floatParam,

float floatDefault,

char * buffer,

int maxlen)

{

snprintf(buffer,maxlen,"%s-%s-%s-%4d-%4d-%5.2f-%5.2f",stringParam, stringDefault, stringParamRw, intParam, intDefault, floatParam, floatDefault);

return buffer;

}

char *

simple_toStr(struct Simple * this,

char * buffer,

int maxlen)

{

snprintf(buffer,maxlen,"%s-%s-%s-%4d-%4d-%5.2f-%5.2f",this->stringParam,this->stringDefault,this->stringParamRw,this->intParam,this->intDefault,this->floatParam,this->floatDefault);

return buffer;

}

char *

simple_toStr_v(void * this,

char * buffer,

int length)

{

return simple_toStr((struct Simple*) this,buffer,length);

}

unsigned long

simple_getHash(struct Simple * this)

{

return pgutil_hash(this->stringParam) +

pgutil_hash(this->stringDefault) +

pgutil_hash(this->stringParamRw) +

this->intParam +

this->intDefault +

this->floatParam +

this->floatDefault;;

}

unsigned long

simple_getHash_v(void * this)

{

return simple_getHash((struct Simple*) this);

}

int

simple_isLessThan(struct Simple * lhs,

struct Simple * rhs)

{

if(strcmp(lhs->stringParam , rhs->stringParam)<0) return 1;

if(strcmp(rhs->stringParam , lhs->stringParam)>0) return 0;

if(strcmp(lhs->stringDefault , rhs->stringDefault)<0) return 1;

if(strcmp(rhs->stringDefault , lhs->stringDefault)>0) return 0;

if(strcmp(lhs->stringParamRw , rhs->stringParamRw)<0) return 1;

if(strcmp(rhs->stringParamRw , lhs->stringParamRw)>0) return 0;

if(lhs->intParam < rhs->intParam) return 1;

if(rhs->intParam > lhs->intParam) return 0;

if(lhs->intDefault < rhs->intDefault) return 1;

if(rhs->intDefault > lhs->intDefault) return 0;

if(lhs->floatParam < rhs->floatParam) return 1;

if(rhs->floatParam > lhs->floatParam) return 0;

if(lhs->floatDefault < rhs->floatDefault) return 1;

if(rhs->floatDefault > lhs->floatDefault) return 0;

return 0;

}

int

simple_isLessThan_v(void * lhs,

void * rhs)

{

return simple_isLessThan((struct Simple*) lhs, (struct Simple* ) rhs);

}

int

simple_isEqualTo(struct Simple * lhs,

struct Simple * rhs)

{

return (strcmp(lhs->stringParam,rhs->stringParam)==0) &&

(strcmp(lhs->stringDefault,rhs->stringDefault)==0) &&

(strcmp(lhs->stringParamRw,rhs->stringParamRw)==0) &&

(lhs->intParam == rhs->intParam) &&

(lhs->intDefault == rhs->intDefault) &&

(lhs->floatParam == rhs->floatParam) &&

(lhs->floatDefault == rhs->floatDefault);

}

int

simple_isEqualTo_v(void * lhs,

void * rhs)

{

return simple_isEqualTo((struct Simple* ) lhs, (struct Simple*) rhs);

}

int

simple_isGreaterThan(struct Simple * lhs,

struct Simple * rhs)

{

if(lhs->floatDefault > rhs->floatDefault) return 1;

if(rhs->floatDefault < lhs->floatDefault) return 0;
