module ADemoClass where

import Text.Printf
import Data.Map
import Data.Set

data ADemoClass = ADemoClass {
  stringParam :: String,
  stringDefault :: String,
  stringParamRw :: String,
  intParam :: Int,
  intDefault :: Int,
  floatParam :: Float,
  floatDefault :: Float
} deriving(Eq,Ord)

new :: String -> String -> Int -> Float -> ADemoClass
new _stringParam _stringParamRw _intParam _floatParam = ADemoClass
  _stringParam
  "string default"
  _stringParamRw
  _intParam
  0
  _floatParam
  0.0



getKey :: String -> String -> String -> Int -> Int -> Float -> Float -> String
getKey stringParam stringDefault stringParamRw intParam intDefault floatParam floatDefault=
        printf "%s-%s-%s-%4d-%4d-%f-%f" stringParam stringDefault stringParamRw intParam intDefault floatParam floatDefault

instance Show ADemoClass where
  show aDemoClass =
    ADemoClass.getKey (stringParam aDemoClass)
           (stringDefault aDemoClass)
           (stringParamRw aDemoClass)
           (intParam aDemoClass)
           (intDefault aDemoClass)
           (floatParam aDemoClass)
           (floatDefault aDemoClass)
