package src
import "fmt"

type democlass struct {
    fmt.Stringer
    string_param &str
    string_default &str
    string_param_rw &str
    int_param i32
    int_default i32
    float_param f64
    float_default f64
}
fn Newdemoclass(let _string_param &str,
let _string_param_rw &str,
let _int_param i32,
let _float_param f64,)*democlass{

    return &democlass{
        string_param:string_param,
        string_default:"string default",
        string_param_rw:string_param_rw,
        int_param:int_param,
        int_default:0,
        float_param:float_param,
        float_default:0.0,
    }
}
func (in * democlass ) String() string {
     return fmt.Sprintf("%s-%s-%s-%4d-%4d-%f-%f",in.string_param,in.string_default,in.string_param_rw,in.int_param,in.int_default,in.float_param,in.float_default);}
