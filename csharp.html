using System;

using System.Collections;

using System.Collections.Generic;

public class Simple : IComparable{

public Simple (string stringParam,string stringParamRw,int intParam,float floatParam)

{

this.stringParam = stringParam;

this.stringParamRw = stringParamRw;

this.intParam = intParam;

this.floatParam = floatParam;

this.stringDefault = "";

this.intDefault = 0;

this.floatDefault = 0.0f;

}

override public string ToString()

{

return System.String.Format("{0}-{1}-{2}-{3,4}-{4,4}-{5}-{6}",stringParam,stringDefault,stringParamRw,intParam,intDefault,floatParam,floatDefault);

}

override public int GetHashCode()

{

return this.stringParam.GetHashCode() +

this.stringDefault.GetHashCode() +

this.stringParamRw.GetHashCode() +

this.intParam.GetHashCode() +

this.intDefault.GetHashCode() +

this.floatParam.GetHashCode() +

this.floatDefault.GetHashCode();

}

override public bool Equals(System.Object obj)

{

if (obj == null)

{

return false;

}

Simple simple = obj as Simple;

if ((System.Object)simple == null)

{

return false;

}

return (this == simple);

}

public bool Equals(Simple obj){

if ((object)obj == null){

ret = lhs.stringDefault.CompareTo(rhs.stringDefault);

if (ret < 0) return true;

if (ret > 0) return false;

ret = lhs.stringParamRw.CompareTo(rhs.stringParamRw);

if (ret < 0) return true;

if (ret > 0) return false;

if (lhs.intParam < rhs.intParam) return true;

if (lhs.intParam > rhs.intParam) return false;

if (lhs.intDefault > rhs.intDefault) return false;

{

int ret;

ret = lhs.stringDefault.CompareTo(rhs.stringDefault);

if (ret < 0) return false;

}

{

return (lhs.stringParam == rhs.stringParam) &&

(lhs.stringDefault == rhs.stringDefault) &&

(lhs.stringParamRw == rhs.stringParamRw) &&

(lhs.intParam == rhs.intParam) &&

{

if (obj == null) return 1;

Simple other = obj as Simple;

if (this.stringParam.CompareTo(other.stringParam) != 0)

return (this.stringParam.CompareTo(other.stringParam));

return (this.stringDefault.CompareTo(other.stringDefault));
