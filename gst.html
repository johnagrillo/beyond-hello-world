ADemoClass.st

Object subclass: ADemoClass[
    |
        stringParam
        stringDefault
        stringParamRw
        intParam
        intDefault
        floatParam
        floatDefault
    |
    ADemoClass class [
        new: stringParam_ stringParamRw:stringParamRw_ intParam:intParam_ floatParam:floatParam_ [
            ^ ((super new) init: stringParam_ stringParamRw:stringParamRw_ intParam:intParam_ floatParam:floatParam_ )
        ]
    ]

    init: stringParam_ stringParamRw:stringParamRw_  intParam:intParam_  floatParam:floatParam_ [
        stringParam:= stringParam_.
        stringParamRw:= stringParamRw_.
        intParam:= intParam_.
        floatParam:= floatParam_.
        stringDefault:= 'string default'.
        intDefault:= 0.
        floatDefault:= 0.0.
    ]
    displayOn: stream [
        <category: 'printing'>
        stringParam displayOn:stream.
        '-' displayOn:stream.
        stringDefault displayOn:stream.
        '-' displayOn:stream.
        stringParamRw displayOn:stream.
        '-' displayOn:stream.
        intParam displayOn:stream.
        '-' displayOn:stream.
        intDefault displayOn:stream.
        '-' displayOn:stream.
        floatParam displayOn:stream.
        '-' displayOn:stream.
        floatDefault displayOn:stream
    ]
]
