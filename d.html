module src.democlass;
import std.string;
import std.string;
import std.typecons;
class Democlass {
public:

  this (string stringParam,string stringParamRw,int intParam,float floatParam)
  {
    this.m_stringParam = stringParam;
    this.m_stringParamRw = stringParamRw;
    this.m_intParam = intParam;
    this.m_floatParam = floatParam;
    this.m_stringDefault = "string default";
    this.m_intDefault = 0;
    this.m_floatDefault = 0.0;
  }

  @property stringParam() pure {return m_stringParam;}
  @property stringParamRw() pure {return m_stringParamRw;}
  @property stringParamRw(string stringParamRw) { m_stringParamRw=stringParamRw;}

  @property intParam() pure {return m_intParam;}
  @property floatParam() pure {return m_floatParam;}


    static string getKey(string stringParam,
                         string stringDefault,
                         string stringParamRw,
                         int intParam,
                         int intDefault,
                         float floatParam,
                         float floatDefault)
    {
        return std.string.format("%s-%s-%s-%4s-%4s-%s-%s",stringParam,stringDefault,stringParamRw,intParam,intDefault,floatParam,floatDefault);
    }

    override string toString()
    {
        return(Democlass.getKey(m_stringParam,m_stringDefault,m_stringParamRw,m_intParam,m_intDefault,m_floatParam,m_floatDefault));
    }

    override hash_t toHash()
    {
    return typeid(m_stringParam).getHash(&m_stringParam) +
        typeid(m_stringDefault).getHash(&m_stringDefault) +
        typeid(m_stringParamRw).getHash(&m_stringParamRw) +
        this.m_intParam +
        this.m_intDefault +
        cast(int)m_floatParam +
        cast(int)m_floatDefault;
    }

    override int opCmp(Object rhs_object)
    {
Democlass rhs = cast(Democlass) rhs_object;
  if (this.m_stringParam < rhs.m_stringParam){
    return -1;
  }
  else if (this.m_stringParam > rhs.m_stringParam){
    return 1;
  }
  else if (this.m_stringDefault < rhs.m_stringDefault){
    return -1;
  }
  else if (this.m_stringDefault > rhs.m_stringDefault){
    return 1;
  }
  else if (this.m_stringParamRw < rhs.m_stringParamRw){
    return -1;
  }
  else if (this.m_stringParamRw > rhs.m_stringParamRw){
    return 1;
  }
  else if (this.m_intParam < rhs.m_intParam){
    return -1;
  }
  else if (this.m_intParam > rhs.m_intParam){
    return 1;
  }
  else if (this.m_intDefault < rhs.m_intDefault){
    return -1;
  }
  else if (this.m_intDefault > rhs.m_intDefault){
    return 1;
  }
  else if (this.m_floatParam < rhs.m_floatParam){
    return -1;
  }
  else if (this.m_floatParam > rhs.m_floatParam){
    return 1;
  }
  else if (this.m_floatDefault < rhs.m_floatDefault){
    return -1;
  }
  else if (this.m_floatDefault > rhs.m_floatDefault){
    return 1;
  }
return 0;

    }
    override bool opEquals(Object rhs_object)
    {
        Democlass rhs = cast(Democlass) rhs_object;
  return (this.m_stringParam == rhs.m_stringParam) && 
         (this.m_stringDefault == rhs.m_stringDefault) && 
         (this.m_stringParamRw == rhs.m_stringParamRw) && 
         (this.m_intParam == rhs.m_intParam) && 
         (this.m_intDefault == rhs.m_intDefault) && 
         (this.m_floatParam == rhs.m_floatParam) && 
         (this.m_floatDefault == rhs.m_floatDefault);
    }

private:
    string m_stringParam;
    string m_stringDefault;
    string m_stringParamRw;
    int m_intParam;
    int m_intDefault;
    float m_floatParam;
    float m_floatDefault;
}
