package src

import "fmt"

type Simple struct {

fmt.Stringer

stringParam string

stringDefault string

stringParamRw string

intParam int

intDefault int

floatParam float64

floatDefault float64

}

func NewSimple(stringParam string,

stringParamRw string,

intParam int,

floatParam float64,)*Simple{

return &Simple{

stringParam:stringParam,

stringDefault:"string default",

stringParamRw:stringParamRw,

intParam:intParam,

intDefault:0,

floatParam:floatParam,

floatDefault:0.0,

}

}

func (in * Simple ) String() string {

return fmt.Sprintf("%s-%s-%s-%4d-%4d-%f-%f",in.stringParam,in.stringDefault,in.stringParamRw,in.intParam,in.intDefault,in.floatParam,in.floatDefault);}
