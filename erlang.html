democlass.erl

%-*-erlang-*-
-module(democlass).
-include("democlass.hrl").

-export([
new/4,
to_string/1
]).

  new(StringParam,
      StringParamRw,
      IntParam,
      FloatParam) -> 
#democlass{    string_param = StringParam,
    string_param_rw = StringParamRw,
    int_param = IntParam,
    float_param = FloatParam}.


to_string (This) ->
    lists:flatten(
        io_lib:format("~s,-~s,-~s,-~4B-~4B-~f-~f",[This#democlass.string_param,This#democlass.string_default,This#democlass.string_param_rw,This#democlass.int_param,This#democlass.int_default,This#democlass.float_param,This#democlass.float_default])).
