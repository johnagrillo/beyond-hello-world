a_demo_class.erl

%-*-erlang-*-
-module(a_demo_class).
-include("a_demo_class.hrl").

-export([
new/4,
to_string/1
]).

  new(StringParam,
      StringParamRw,
      IntParam,
      FloatParam) -> 
#a_demo_class{    string_param = StringParam,
    string_param_rw = StringParamRw,
    int_param = IntParam,
    float_param = FloatParam}.


to_string (This) ->
    lists:flatten(
        io_lib:format("~s,-~s,-~s,-~4B-~4B-~f-~f",[This#a_demo_class.string_param,This#a_demo_class.string_default,This#a_demo_class.string_param_rw,This#a_demo_class.int_param,This#a_demo_class.int_default,This#a_demo_class.float_param,This#a_demo_class.float_default])).
