package src;
import java.util.*;
import org.javatuples.*;

public class ADemoClass
 implements Comparable<ADemoClass>
{

    public ADemoClass(String stringParam,String stringParamRw,Integer intParam,Double floatParam)
    {
        this.stringParam = stringParam;
        this.stringParamRw = stringParamRw;
        this.intParam = intParam;
        this.floatParam = floatParam;
        this.stringDefault = "";
        this.intDefault = 0;
        this.floatDefault = 0.0;
    }
    String getStringParam(){ return this.stringParam;}

    String getStringParamRw(){ return this.stringParamRw;}
    String setStringParamRw(String _stringParamRw){        this.stringParamRw=_stringParamRw;        return this.stringParamRw;}


    Integer getIntParam(){ return this.intParam;}

    Double getFloatParam(){ return this.floatParam;}

    @Override public String toString()
    {
        return String.format("%s-%s-%s-%4d-%4d-%s-%s",stringParam,stringDefault,stringParamRw,intParam,intDefault,floatParam,floatDefault);
    }

    @Override public int hashCode()
    {
        return stringParam.hashCode()+stringDefault.hashCode()+stringParamRw.hashCode()+intParam+intDefault+floatParam.hashCode()+floatDefault.hashCode();
    }
    @Override public boolean equals(Object obj)
    {
        if (this == obj) return true;
        if (!(obj instanceof ADemoClass)) return false;
        ADemoClass aDemoClass = (ADemoClass)obj;
        return stringParam == aDemoClass.stringParam &&
            stringDefault == aDemoClass.stringDefault &&
            stringParamRw == aDemoClass.stringParamRw &&
            intParam == aDemoClass.intParam &&
            intDefault == aDemoClass.intDefault &&
            floatParam == aDemoClass.floatParam &&
            floatDefault == aDemoClass.floatDefault;
    }

    @Override public int compareTo(ADemoClass rhs)
    {
        int ret=0;
        ret = stringParam.compareTo(rhs.stringParam);
        if (ret !=0) return ret;
        ret = stringDefault.compareTo(rhs.stringDefault);
        if (ret !=0) return ret;
        ret = stringParamRw.compareTo(rhs.stringParamRw);
        if (ret !=0) return ret;

        if (intParam < rhs.intParam) return -1;
        if (intParam > rhs.intParam) return 1;


        if (intDefault < rhs.intDefault) return -1;
        if (intDefault > rhs.intDefault) return 1;


        if (floatParam < rhs.floatParam) return -1;
        if (floatParam > rhs.floatParam) return 1;


        if (floatDefault < rhs.floatDefault) return -1;
        if (floatDefault > rhs.floatDefault) return 1;

        return ret;
    }
    private String stringParam;
    private String stringDefault;
    private String stringParamRw;
    private Integer intParam;
    private Integer intDefault;
    private Double floatParam;
    private Double floatDefault;

}
