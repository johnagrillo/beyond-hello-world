class Democlass(stringParam,stringParamRw,intParam,floatParam)

   _stringDefault = "string default"
   _stringParamRw = stringParamRw
   _intParam = intParam
   _intDefault = 0
   _floatParam = floatParam
   _floatDefault = 0.0

   function __get_stringParam(): return self._stringParam

   function __get_stringParamRw(): return self._stringParamRw
   function __set_stringParamRw(v): self._stringParamRw=v


   function __get_intParam(): return self._intParam

   function __get_floatParam(): return self._floatParam

   function toString()
      return @"$(self._stringParam:4r)-$(self._stringDefault:4r)-$(self._stringParamRw:4r)-$(self._intParam:4r)-$(self._intDefault:4r)-$(self._floatParam:4r)-$(self._floatDefault:4r)"
   end

   function compare(rhs)
      ret = self.stringParam.compare(rhs.stringParam)
      if ret != 0
         return ret
      end
      ret = self.stringDefault.compare(rhs.stringDefault)
      if ret != 0
         return ret
      end
      ret = self.stringParamRw.compare(rhs.stringParamRw)
      if ret != 0
         return ret
      end
      ret = self.intParam.compare(rhs.intParam)
      if ret != 0
         return ret
      end
      ret = self.intDefault.compare(rhs.intDefault)
      if ret != 0
         return ret
      end
      ret = self.floatParam.compare(rhs.floatParam)
      if ret != 0
         return ret
      end
      ret = self.floatDefault.compare(rhs.floatDefault)
      if ret != 0
         return ret
      end

   return ret
   end
end
