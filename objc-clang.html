-(BOOL) isEqual:(ADemoClass * ) _other
{
  return ([stringParam isEqual: _other->stringParam]) && 
        ([stringDefault isEqual: _other->stringDefault]) && 
        ([stringParamRw isEqual: _other->stringParamRw]) && 
        (intParam == _other->intParam) && 
        (intDefault == _other->intDefault) && 
        (floatParam == _other->floatParam) && 
        (floatDefault == _other->floatDefault);
}
-(NSComparisonResult) localizedCompare:(ADemoClass * ) _other
{
    if ([stringParam localizedCompare : _other->stringParam]!=NSOrderedSame){
        return [stringParam localizedCompare : _other->stringParam];
    }
    if ([stringDefault localizedCompare : _other->stringDefault]!=NSOrderedSame){
        return [stringDefault localizedCompare : _other->stringDefault];
    }
    if ([stringParamRw localizedCompare : _other->stringParamRw]!=NSOrderedSame){
        return [stringParamRw localizedCompare : _other->stringParamRw];
    }
    if (intParam < _other->intParam) return NSOrderedAscending;
    if (intParam > _other->intParam) return NSOrderedDescending;
    if (intDefault < _other->intDefault) return NSOrderedAscending;
    if (intDefault > _other->intDefault) return NSOrderedDescending;
    if (floatParam < _other->floatParam) return NSOrderedAscending;
    if (floatParam > _other->floatParam) return NSOrderedDescending;
    if (floatDefault < _other->floatDefault) return NSOrderedAscending;
    if (floatDefault > _other->floatDefault) return NSOrderedDescending;
    return NSOrderedSame;

}
