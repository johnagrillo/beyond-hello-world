new_simple (){

string_param=$1

string_param_rw=$2

int_param=$3

float_param=$4

NEW_SIMPLE=(

["string_param"]=$string_param

["string_param_rw"]=$string_param_rw

["int_param"]=$int_param

["float_param"]=$float_param

["string_default"]="string default"

["int_default"]=0

["float_default"]=0.0

)

}
