simple.clj

(ns src.simple)
(defrecord Simple[string-param string-default string-param-rw int-param int-default float-param float-default]
  java.lang.Comparable

  (toString  [this]
             (format "%s-%s-%4d-%f"
                      (:string-param this)
                      (:string-param-rw this)
                      (:int-param this)
                      (:float-param this)))

  (compareTo [this other]
     (let [func ( fn [obj] (reduce #(conj %1 (%2 obj)) [] (keys obj)))]
        (.compareTo  (func this) (func other))))
)
(defn make-Simple
[string-param string-param-rw int-param float-param]
(src.simple.Simple. 
  string-param
  "string default"
  string-param-rw
  int-param
  0
  float-param
  0.0
)
