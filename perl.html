use strict;

package Simple;

use overload('""', 'toString');

my $STRINGPARAM = "string param";

my $STRINGDEFAULT = "string default";

my $STRINGPARAMRW = "string param rw";

my $INTPARAM = "int param";

my $INTDEFAULT = "int default";

my $FLOATPARAM = "float param";

my $FLOATDEFAULT = "float default";

sub new {

my ($this,$_stringParam,$_stringParamRw,$_intParam,$_floatParam) = @_;

my $ref = ref($this)||$this;

my $self = {};

bless $self,$ref;

$self->{$STRINGPARAM} = $_stringParam;

$self->{$STRINGPARAMRW} = $_stringParamRw;

$self->{$INTPARAM} = $_intParam;

$self->{$FLOATPARAM} = $_floatParam;

$self->{$STRINGDEFAULT} = "string default";

$self->{$INTDEFAULT} = 0;

$self->{$FLOATDEFAULT} = 0.0;

return $self;

}

sub getStringParam {

my ($self) = @_;

$self->{$STRINGPARAM};

}

sub getStringDefault {

my ($self) = @_;

$self->{$STRINGDEFAULT};

}

sub getStringParamRw {

my ($self) = @_;

$self->{$STRINGPARAMRW};

}

sub getIntParam {

my ($self) = @_;

$self->{$INTPARAM};

}

sub getIntDefault {

my ($self) = @_;

$self->{$INTDEFAULT};

}

sub getFloatParam {

my ($self) = @_;

$self->{$FLOATPARAM};

}

sub getFloatDefault {

my ($self) = @_;

$self->{$FLOATDEFAULT};

}

sub setStringDefault {

my ($self,$param) = @_;

$self->{$STRINGDEFAULT} = $param;

}

sub setStringParamRw {

my ($self,$param) = @_;

$self->{$STRINGPARAMRW} = $param;

}

sub setIntDefault {

my ($self,$param) = @_;

$self->{$INTDEFAULT} = $param;

}

sub setFloatDefault {

my ($self,$param) = @_;

$self->{$FLOATDEFAULT} = $param;

}

sub toString {

my ($self) = @_;

sprintf("%s-%s-%s-%s-%s-%s-%s",

$self->{$STRINGPARAM},$self->{$STRINGDEFAULT},$self->{$STRINGPARAMRW},$self->{$INTPARAM},$self->{$INTDEFAULT},$self->{$FLOATPARAM},$self->{$FLOATDEFAULT});

}

1;
