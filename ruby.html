class Simple
  attr_accessor :string_param_rw
  attr_reader :string_param, :int_param, :float_param

  def initialize(string_param,
                 string_param_rw,
                 int_param,
                 float_param)
   @string_param = string_param
   @string_param_rw = string_param_rw
   @int_param = int_param
   @float_param = float_param
   @string_default = "string default"
   @int_default = 0
   @float_default = 0.0
  end

  def self.get_key(string_param,
                   string_default,
                   string_param_rw,
                   int_param,
                   int_default,
                   float_param,
                   float_default)
    "%s-%s-%s-%4d-%4d-%f-%f"%[string_param,string_default,string_param_rw,int_param,int_default,float_param,float_default]
  end

  def to_s()
        Simple.get_key(@string_param,@string_default,@string_param_rw,@int_param,@int_default,@float_param,@float_default)
  end

  def <=>(rhs)
    ret = string_param <=> rhs.string_param
    return ret if ret != 0
    ret = string_default <=> rhs.string_default
    return ret if ret != 0
    ret = string_param_rw <=> rhs.string_param_rw
    return ret if ret != 0
    ret = int_param <=> rhs.int_param
    return ret if ret != 0
    ret = int_default <=> rhs.int_default
    return ret if ret != 0
    ret = float_param <=> rhs.float_param
    return ret if ret != 0
    ret = float_default <=> rhs.float_default
    return ret if ret != 0

  end

  def eql?(rhs)
    string_param.eql?(rhs.string_param) and string_default.eql?(rhs.string_default) and string_param_rw.eql?(rhs.string_param_rw) and int_param.eql?(rhs.int_param) and int_default.eql?(rhs.int_default) and float_param.eql?(rhs.float_param) and float_default.eql?(rhs.float_default)
  end

  def hash()
    string_param.hash + string_default.hash + string_param_rw.hash + int_param.hash + int_default.hash + float_param.hash + float_default.hash
  end


end
